1 /**
  2  * @name ClusterManager
  3  * @version 2.0
  4  * @author Marcus Ulrich
  5  * @fileoverview
  6  * This library creates and manages clusters for Google Maps API v3. It does two things to make maps 
  7  * with large numbers of markers more useable: 1) Combines markers in close proximity to each other 
  8  * based on zoom level into clusters, 2) Only adds markers in the current viewport (and optional 
  9  * padding) to the map.
 10  * <b>How it works</b>:<br/>
 11  * The manager sets up a dictionary for clusters and a dictionary for markers. Every marker that's 
 12  * added to the manager has a string created based on it's latitude, longitude, and zoom level and 
 13  * that's used to add it to the cluster dictionary. Nearby markers will hash to the same string so 
 14  * nothing has to be calculated. Nearby clusters are then combined.
 15  * Markers can be added with optional type and subtypes so subsets of markers can be shown and 
 16  * hidden. Markers with the same subtype will still be clustered together, but can be shown or 
 17  * hidden seperately. Markers with the same type will be clustered together and can also be hidden
 18  * or shown seperately.
 19  * The function used to create the clusters is stored and this function can be overridden for 
 20  * greater control of the look and/or behavior of the clusters for each marker type.
 21  */
 22

 27 /**
 28  * Creates a new Cluster Manager for clustering markers on a V3 Google map.
 29  *
 30  * @param {GMap3} map The map that the markers should be added to.
 31  * @param {object} [opts] Options for configuring the behavior of the clustering. Defaults are 
 32  * applied in resetManager.
 33  * @param {google.maps.Marker[]} [opts.markers] Markers to add to the manager.
 34  * @param {function} [opts.zoom_to_precision=function(zoom_level) {return zoom_level + precision;}] 
 35  * A function to set the precision for each zoom level. 
 36  * @param {number} [opts.precision=2] A number between 0 and 27 that sets how small the cluster 
 37  * boxes will be. Higher numbers will make smaller boxes.
 38  * @param {string|object} [opts.icon_color="00CC00"] Sets the default icon color in HEX. Default is 
 39  * a bright green.
 40  * @param {number} [opts.padding=200] The amount of padding in pixels where markers not in the 
 41  * viewport will still be added to the map.
 42  * @param {boolean} [opts.visualize=false] For debugging. Will put a box around each cluster with at 
 43  * least one marker.
 44  * @param {number} [opts.cluster_by_distance=true] Combine neighboring clusters if they are close 
 45  * together. This is a little slower but makes more rounded clusters.
 46  * @param {number} [opts.cluster_distance_factor=2048000] Clusters are combined if they are within 
 47  * this distance: cluster_distance_factor*Math.pow(2, -precision+2)
 48  * @constructor
 49  */
 50 ClusterManager = function(map, opts) {
 51     var me = this;
 52     opts = opts || {};
 53     this.map = map;
 54     this.setMap(map);
 55     this.resetManager(opts);
 56     this.setPrecision(this.zoomToPrecision(this.map.getZoom()));
 57     google.maps.event.addDomListener(map, "dragstart", function() {
 58         me.mapDragging = true;
 59     });
 60     google.maps.event.addDomListener(map, "dragend", function() {
 61         me.mapDragging = false;
 62         me._onMapMoveEnd();
 63     });
 64     google.maps.event.addDomListener(map, "center_changed", function() {
 65         if (!me.mapDragging) me._onMapMoveEnd();
 66     });
 67     google.maps.event.addDomListener(map, "zoom_changed", function() {
 68         me._onMapMoveEnd();
 69     });
 70     if (typeof opts.markers !== "undefined") this.addMarkers(opts.markers);
 71 };

 89 /**
 90  * Sets the marker and clusters back to the inital state.
 91  *
 92  * @param {object} [opts] Options for configuring the behavior of the clustering. Defaults are 
 93  * applied in resetManager.
 94  * @param {function} [opts.zoom_to_precision=function(zoom_level) {return zoom_level + precision;}] 
 95  * A function to set the precision for each zoom level. 
 96  * @param {number} [opts.precision=2] A number between 0 and 27 that sets how small the cluster 
 97  * boxes will be. Higher numbers will make smaller boxes.
 98  * @param {string|object} [opts.icon_color="00CC00"] Sets the default icon color in HEX. Default is 
 99  * a bright green.
100  * @param {number} [opts.padding=200] The amount of padding in pixels where markers not in the 
101  * viewport will still be added to the map.
102  * @param {boolean} [opts.visualize=false] For debugging. Will put a box around each cluster with at 
103  * least one marker.
104  * @param {number} [opts.cluster_by_distance=true] Combine neighboring clusters if they are close 
105  * together. This is a little slower but makes more rounded clusters.
106  * @param {number} [opts.cluster_distance_factor=2048000] Clusters are combined if they are within 
107  * this distance: cluster_distance_factor*Math.pow(2, -precision+2)
108  */
109 ClusterManager.prototype.resetManager = function(opts) {
110     this.markers = {}; //hold markers by type, then subtype.
111     this.clusters = {}; //define clusters by precision, type, then geobox.
112     this.cluster_fns = {}; //store cluster function for building the cluster markers.
113     this.cluster_meta = {}; //marker counts, etc
114     var precision = opts.precision >= 0 && opts.precision <= 27 ? opts.precision:2;
115     opts = ClusterManager.applyDefaults({
116         padding                 : 200,
117         visualize               : false,
118         zoom_to_precision       : function(zoom_level) {
119             return zoom_level + precision;
120         },
121         cluster_by_distance     : true,
122         cluster_distance_factor : 2048000,
123         icon_color              : "00CC00"
124     }, opts);
125     this.opts = opts;
126 };

128 /**
129  * Sets the current level of precision.
130  * To speed up clustering and reduce memory, only the clusters for the current precision are 
131  * calculated so changing the precision may take extra time to calculate clusters at the new 
132  * precision.
133  *
134  * @param {number} precision The level to set the precision to. Currently, must be from 1 to 49.
135  * @private
136  */
137 ClusterManager.prototype.setPrecision = function(precision) {
138     if(precision >= 50 || precision < 0) return;
139     this.current_precision_ = precision;
140     this.clear();
141     if (typeof this.clusters[precision] === "undefined") {
142         var markers = this.getMarkers();
143         for(var i=0, length=markers.length; i<length; i++) { 
144             var marker = markers[i];
145             if (this.getMarkerMeta(marker).subtype !== "cluster") {
146                 this.addToCluster(marker, this.getMarkerMeta(marker).type, precision);
147             }
148         }
149     }
150     this.cluster();
151     this.updateMarkers();
152 };

212  * 40	5.10993e-10	0.00005688358228815859m
213  * 41	2.54999e-10	0.000028386423065207123m
214  * 42	1.27016e-10	0.000014139425398842023m
215  * 43	6.30109e-11	0.00000701434462054884m
216  * 44	3.10080e-11	0.0000034518042314022482m
217  * 45	1.50066e-11	0.0000016705340368289525m
218  * 46	6.99174e-12	7.783169944316711e-7m
219  * 47	3.01270e-12	3.353723634542973e-7m
220  * 48	9.94759e-13	1.1073615774434343e-7m
221  * 
222  * @param {number} lat Latitude. Value is clamped to the nearest value in [-90.0, 90.0];
223  * @param {number} lng Longitude. Value is wrapped to stay within [-180, 180);
224  * @param {number} precision An integer representing the number of bits to take from the 
225  *                           untruncated latitude and longitude hashes.
226  * @returns {string} geohash A binary hash string with a length twice the precision.
227  */
228 ClusterManager.prototype.getGeohash = function(lat, lng, precision) {
229     lat = Math.min(lat, 90.0);
230     lat = Math.max(lat, -90.0);
231     lng = Math.abs((lng+180.0)%360.0) - 180.0;
232 
233     if (precision <= 0) return "";
234     var max_power = 12 //This is the limit for maximum range of decimal numbers in javascript.
235     // Make the latitude and longitude positive and then mulitiply them by 10^12 to get rid of
236     // as many decimal places as possible. Then change this to binary.
237     var latBase = parseInt((lat + 90.0) * (Math.pow(10, max_power))).toString(2);
238     var lngBase = parseInt((lng + 180.0) * (Math.pow(10, max_power))).toString(2);
239     //Pad the front with zeros to make sure latitude and longitude are 49 bits.
240     var fortyninezeros = "0000000000000000000000000000000000000000000000000";
241     var latHash = fortyninezeros.substr(0, 49 - latBase.length) + latBase;
242     var lngHash = fortyninezeros.substr(0, 49 - lngBase.length) + lngBase;
243     //Take bits from the front based on the precision. 
244     //Concatinate the latitude and longitude strings.
245     var geohash = latHash.substr(0, precision) + lngHash.substr(0, precision);
246     return geohash;
247 };

249 /**
250  * Given a geohash, this returns the bounds on it's range. The inverse of getGeohash.
251  * 
252  * @param {string} geohash A string representing the geobox.
253  * @returns {google.maps.LatLngBounds} The bounds on the geobox. 
254  */
255 ClusterManager.prototype.geohashGetLatLngBounds = function(geohash) {
256     var max_power = 12;
257     var precision = this.geohashGetPrecision(geohash);
258     var fortyninezeros = "0000000000000000000000000000000000000000000000000";
259     var latMinHashBin = geohash.substr(0, precision) + fortyninezeros.substr(0, 49 - precision);
260     var lngMinHashBin = geohash.substr(precision, geohash.length) 
261                       + fortyninezeros.substr(0, 49 - precision);
262     var fortynineones = "1111111111111111111111111111111111111111111111111";
263     var latMaxHashBin = geohash.substr(0, precision) + fortynineones.substr(0, 49 - precision);
264     var lngMaxHashBin = geohash.substr(precision, geohash.length) 
265                       + fortynineones.substr(0, 49 - precision);
266     var latMinHashDec = parseInt(latMinHashBin, 2);
267     var lngMinHashDec = parseInt(lngMinHashBin, 2);
268     var latMaxHashDec = parseInt(latMaxHashBin, 2);
269     var lngMaxHashDec = parseInt(lngMaxHashBin, 2);
270     var latMin = Math.max(-90.0,  (latMinHashDec / Math.pow(10, max_power)) - 90);
271     var lngMin = Math.max(-180.0, (lngMinHashDec / Math.pow(10, max_power)) - 180);
272     var latMax = Math.min(90.0,   (latMaxHashDec / Math.pow(10, max_power)) - 90);
273     var lngMax = Math.min(180.0,  (lngMaxHashDec / Math.pow(10, max_power)) - 180);
274     return new google.maps.LatLngBounds(new google.maps.LatLng(latMin, lngMin), 
275                                         new google.maps.LatLng(latMax, lngMax));
276 };

278 /**
279  * Derives the precision from a geohash string.
280  *
281  * @param {string} geohash The geohash to find the precision of.
282  * @returns {number} The derived precision of the geobox.
283  * @private
284  */
285 ClusterManager.prototype.geohashGetPrecision = function(geohash) {
286     var precision = geohash.length / 2;
287     if (parseInt(precision) !== precision || precision < 0 || precision >= 50) return undefined;
288     return precision;
289 };

291 /**
292  * Gets the boxes surrounding the given box and only returns boxes that have at least one marker.
293  *
294  * @param {string} box_str The geobox to find the neighbors of.
295  * @param {string} type The type of the geobox to find the neighbors of.
296  * @returns {string[]} The strings for the geoboxes with at least one marker neighboring the input 
297  * geobox.
298  * @private
299  */
300 ClusterManager.prototype.getNeighborBoxes = function(box_str, type) {
301     var bounds = this.geohashGetLatLngBounds(box_str);
302     var precision = this.geohashGetPrecision(box_str);
303     var boxString1 = this.getGeohash(bounds.getSouthWest().lat() + 0.0001, 
304                                      bounds.getSouthWest().lng() - 0.0001, precision);
305     var boxString2 = this.getGeohash(bounds.getSouthWest().lat() - 0.0001, 
306                                      bounds.getSouthWest().lng() + 0.0001, precision);
307     var boxString3 = this.getGeohash(bounds.getNorthEast().lat() + 0.0001, 
308                                      bounds.getNorthEast().lng() - 0.0001, precision);
309     var boxString4 = this.getGeohash(bounds.getNorthEast().lat() - 0.0001, 
310                                      bounds.getNorthEast().lng() + 0.0001, precision);
311     var boxString5 = this.getGeohash(bounds.getSouthWest().lat() + 0.0001, 
312                                      bounds.getSouthWest().lng() + 0.0001, precision);
313     var boxString6 = this.getGeohash(bounds.getSouthWest().lat() - 0.0001, 
314                                      bounds.getSouthWest().lng() - 0.0001, precision);
315     var boxString7 = this.getGeohash(bounds.getNorthEast().lat() + 0.0001, 
316                                      bounds.getNorthEast().lng() + 0.0001, precision);
317     var boxString8 = this.getGeohash(bounds.getNorthEast().lat() - 0.0001, 
318                                      bounds.getNorthEast().lng() - 0.0001, precision);
319     var boxStrings = [boxString1, boxString2, boxString3, boxString4, boxString5, boxString6, 
320                       boxString7, boxString8];
321     for (var i = 0, neighbors = [], boxString; boxString = boxStrings[i]; i++) {
322         if (typeof this.clusters[precision][type][boxString] !== "undefined" 
323             && boxString !== box_str) {
324             neighbors.push(boxString);
325         }
326     }
327     return neighbors;
328 };

335  * @param {object} [opts] Options for the appearance of the polygon.

339  * @param {string} [opts.strokeOpacity]

341  * @param {string} [opts.fillOpacity] .
342  * @returns {google.maps.Polygon} A polygon covering the box's bounds.
343  */
344 ClusterManager.prototype.boxToPolygon = function(geohash, opts) {
345     opts = ClusterManager.applyDefaults({
346         map           : this.map,
347         strokeColor   : "#f33f00",
348         strokeWeight  : 5,
349         strokeOpacity : 1,
350         fillColor     : "#ff0000",
351         fillOpacity   : 0.2
352     }, opts);
353     var bounds = this.geohashGetLatLngBounds(geohash);  //TODO:change back!!

355     var sw = bounds.getSouthWest();
356     var polygon = new google.maps.Polygon({
357         paths         : opts.paths || [ne, new google.maps.LatLng(ne.lat(), sw.lng()), sw,

362         fillColor     : opts.fillColor,
363         fillOpacity   : opts.fillOpacity,
364         map           : opts.map
365     });
366     return polygon;

369 /**
370  * Tests whether a geobox touches a given bounds. Padding expands the range of the bounds based on 
371  * viewport pixels.
372  *
373  * @param {string} geohash A string representing the geobox.
374  * @param {google.maps.LatLngBounds} bounds The bounds to be tested.
375  * @param {number} [padding] The number of pixels to expand the bounds. 
376  * @returns {boolean} True if any part of the geobox touches the bounds expanded by the padding.
377  * @private
378  */
379 ClusterManager.prototype.boxInBounds = function(geohash, bounds, padding) {
380     //make a new LatLngBounds so we don't have any side effects on our map bounds.
381     var newBounds = new google.maps.LatLngBounds(this.map.getBounds().getSouthWest(), 
382                                                  this.map.getBounds().getNorthEast());
383     if (typeof padding !== "undefined") {
384         var proj = this.map.getProjection();
385         var scale = Math.pow(2, this.map.getZoom());
386         var pixelOffset = new google.maps.Point((padding / scale) || 0, (padding / scale) || 0);
387         var nePoint = proj.fromLatLngToPoint(bounds.getNorthEast());
388         var swPoint = proj.fromLatLngToPoint(bounds.getSouthWest());
389         var newNEPoint = new google.maps.Point(nePoint.x + pixelOffset.x, 
390                                                nePoint.y - pixelOffset.y);
391         var newSWPoint = new google.maps.Point(swPoint.x - pixelOffset.x, 
392                                                swPoint.y + pixelOffset.y);
393         var newNE = proj.fromPointToLatLng(newNEPoint);
394         var newSW = proj.fromPointToLatLng(newSWPoint);
395         newBounds.extend(newNE);
396         newBounds.extend(newSW);
397     }
398     var boxBounds = this.geohashGetLatLngBounds(geohash);
399     if (newBounds.contains(boxBounds.getNorthEast()) || 
400         newBounds.contains(boxBounds.getSouthWest()) || 
401         boxBounds.toSpan().lat() == 180) return true;
402     else return false;
403 };

405 /**
406  * Use this to add markers in one batch through an array.
407  *
408  * @param {google.maps.Marker[]} markers An array of markers.
409  * @param {string} type The type for the markers being added.
410  * @param {string} subtype The subtype for the markers being added.
411  */
412 ClusterManager.prototype.addMarkers = function(markers, type, subtype) {
413     if (Object.prototype.toString.call(markers) === '[object Array]') {
414 
415         for(var i=0, length=markers.length; i<length; i++) { 
416             var marker = markers[i];
417             this.addMarker(marker, {
418                 "type"    : type,
419                 "subtype" : subtype
420             });
421         }
422     }
423 };

425 /**
426  * Add a single marker to the map. Stores an associative array for looking for marker types so we 
427  * can cluster by type. Doesn't build clusters or add them to the map. Each marker can have an opt 
428  * type and subtype to cluster by. 
429  *
430  * @param {google.maps.Marker} marker The marker to add. 
431  * @param {object} [opts] Options for the behavior of the marker in the clusters.
432  * @param {string} [opts.type] A string that is used to sort which markers to cluster.
433  * @param {string} [opts.subtype] A string that is used to show/hide subsets of markers of a given 
434  * type.
435  * @param {boolean} [opts.hidden] Set true to make a marker disappear from the map even if it's in 
436  * the viewport.
437  * @param {boolean} [opts.visible] Set true if the marker is visible in the viewport. 
438  * @param {string} [opts.summary] The summary text that appears in the cluster's infowindow. 
439  * Clicking on the text opens the markers infowindow.
440  */
441 ClusterManager.prototype.addMarker = function(marker, opts) {
442     if (typeof opts === "undefined") opts = this.getMarkerMeta(marker);
443     //Set when the marker is visible in the viewport and not hidden.
444     //Set when we want to hide the marker even if it's in the viewport.
445     var defaults = {
446         type    : "generic",
447         subtype : "generic",
448         hidden  : true,
449         visible : false
450     };
451     opts = ClusterManager.applyDefaults(defaults, opts);
452     var type = opts.type,
453         subtype = opts.subtype;
454     //if this is the first marker of the type, save the cluster function.
455     if (typeof this.markers[type] === "undefined") {
456         this.markers[type] = {};
457         this.cluster_meta[type] = {
458             count: {
459                 total   : 0,
460                 visible : 0,
461                 cluster : 0
462             }
463         };
464     }
465     if (typeof this.cluster_fns[type] === "undefined") {
466         this.setClusterFn(type, this.createClusterMarker);
467     }
468     //if this is the first marker of the subtype, set up an empty array to save it in.
469     if (typeof this.markers[type][subtype] === "undefined") {
470         this.markers[type][subtype] = [];
471     }
472     this.markers[type][subtype].push(marker);
473     if (subtype !== "cluster") {
474         this.cluster_meta[type]["count"]["total"] += 1;
475         this.addToCluster(marker, type, this.getPrecision());
476     }
477     if (typeof opts.summary === "undefined") {
478         var capType = opts.type.charAt(0).toUpperCase() + opts.type.slice(1);
479         opts.summary = typeof marker.getTitle() === "undefined" ? capType + " marker " 
480                      + this.count(opts.type, "total") : marker.getTitle();
481     }
482     this.setMarkerMeta(marker, opts);
483 };

495 /**
496  * Adds a marker to a cluster object. Does not create the cluster markers.
497  *
498  * @param {google.maps.Marker} marker The marker to add. 
499  * @param {string} type The type of the marker to add. This will be used to form cluster groups. If 
500  * no type is given it is assigned type "generic".
501  * @param {number} precision The precision to cluster at.
502  * @param {string} [geohash] Force a marker into a particular geobox rather than its default one.
503  * @private
504  */
505 ClusterManager.prototype.addToCluster = function(marker, type, precision, geohash) {
506     var clusters = this.clusters;
507     var markerLL = marker.getPosition();
508     var markerLat = markerLL.lat();
509     var markerLng = markerLL.lng();
510     if (typeof clusters[precision] === "undefined") {
511         clusters[precision] = {};
512     }
513     if (typeof clusters[precision][type] === "undefined") {
514         clusters[precision][type] = {};
515     }
516     var cluster = clusters[precision][type];
517     if (typeof geohash === "undefined") {
518         geohash = this.getGeohash(markerLat, markerLng, precision);
519     }
520     if (typeof cluster[geohash] !== "undefined") {
521         cluster[geohash]["markers"].push(marker);
522         var length = cluster[geohash]["markers"].length;
523         var lat = ((length - 1) / length) * cluster[geohash]["center"][0] + markerLat / length;
524         var lng = ((length - 1) / length) * cluster[geohash]["center"][1] + markerLng / length;
525         cluster[geohash]["center"] = [lat, lng];
526     } else {
527         cluster[geohash] = {
528             cluster : false,
529             markers : [marker],
530             center  : [markerLat, markerLng]
531         };
532     }
533 };

535 /**
536  * Removes a marker from a cluster and resets the cluster box's properties.
537  *
538  * @param {google.maps.Marker} marker The marker to remove.
539  * @param {string} geohash The geohash to remove the marker from.
540  * @private
541  */
542 ClusterManager.prototype.removeFromCluster = function(marker, geohash) {
543     var precision = this.geohashGetPrecision(geohash);
544     var type = this.getMarkerMeta(marker).type;
545     var geoBox = this.clusters[precision][type][geohash];
546     if (geoBox["markers"].length === 1) {
547         delete(this.clusters[precision][type][geohash]);
548     } else if (geoBox["markers"].length > 1) {
549         for (var i=0, new_markers=[], center_lat=0, center_lng=0, test_marker; 
550              test_marker = geoBox["markers"][i]; i++) {
551             if (test_marker !== marker) {
552                 new_markers.push(test_marker);
553                 center_lat = center_lat + test_marker.getPosition().lat();
554                 center_lng = center_lng + test_marker.getPosition().lng();
555             }
556         }
557         center_lat = center_lat / new_markers.length;
558         center_lng = center_lng / new_markers.length;
559         geoBox["center"] = [center_lat, center_lng];
560         geoBox["markers"] = new_markers;
561         geoBox["cluster"] = false;
562         this.clusters[precision][type][geohash] = geoBox;
563     }
564 };

566 /**
567  * This takes two geoboxes and puts all the markers into the one with more markers or the first one.
568  * 
569  * @param {string} box_str1 First box to combine.
570  * @param {string} box_str2 Second box to combine.
571  * @param {string} type Type of the boxes since this can't be derived.
572  * @private
573  */
574 ClusterManager.prototype.combineBoxes = function(box_str1, box_str2, type) {
575     var precision = this.geohashGetPrecision(box_str1);
576     if (this.clusters[precision][type][box_str1]["markers"].length < 
577         this.clusters[precision][type][box_str2]["markers"].length) {
578         var temp = box_str1;
579         box_str1 = box_str2;
580         box_str2 = temp;
581     }
582     var length = this.clusters[precision][type][box_str2]["markers"].length;
583     for (var i = length - 1, marker; i >= 0; i--) {
584         marker = this.clusters[precision][type][box_str2]["markers"][i];
585         this.removeFromCluster(marker, box_str2);
586         this.addToCluster(marker, type, precision, box_str1);
587     }
588 };

590 /**
591  * This checks neighboring geoboxes to see if they are centered within a minimum distance. This 
592  * makes the clusters less box shaped, but also takes extra time.
593  * 
594  * @param {string} type The type of the markers to cluster.
595  * @private
596  */
597 ClusterManager.prototype.combineClustersByDistance = function(type) {
598     var precision = this.getPrecision();
599     var clusters = this.clusters;
600     var clusterDistanceFactor = this.opts.cluster_distance_factor || 2048000;
601     for (var boxStr in clusters[precision][type]) {
602         var neighbors = this.getNeighborBoxes(boxStr, type);
603         var distance = clusterDistanceFactor * Math.pow(2, -precision + 2);
604         var clusterCenter = clusters[precision][type][boxStr]["center"];
605 /***
606         new google.maps.Circle({
607                 strokeColor   : '#FF0000',
608                 strokeOpacity : 0.8,
609                 strokeWeight  : 2,
610                 fillColor     : '#FF0000',
611                 fillOpacity   : 0.35,
612                 map           : this.map,
613                 center        : new google.maps.LatLng(clusterCenter[0], clusterCenter[1]),
614                 radius        : distance});
615 ***/
616         for (var j = 0, result = 0, neighborStr; neighborStr = neighbors[j]; j++) {
617             var clusterCenter = clusters[precision][type][boxStr]["center"];
618             var neighborCenter = clusters[precision][type][neighborStr]["center"];
619             var currentDist = google.maps.geometry.spherical.computeDistanceBetween(
620                               new google.maps.LatLng(clusterCenter[0], clusterCenter[1]), 
621                               new google.maps.LatLng(neighborCenter[0], neighborCenter[1]));
622             if (currentDist < distance) {
623                 result = j;
624                 distance = currentDist;
625             }
626         }
627         if (result) {
628             neighborStr = neighbors[result];
629             this.combineBoxes(boxStr, neighborStr, type);
630         }
631     }
632 };

634 /**
635  * This builds the actual cluster markers and optionally combines boxes if the markers get too close 
636  * together. It does not set up the cluster dictionary.
637  *
638  * @param {string} [type] The type to cluster. If none is given, this sets up the clusters for every 
639  * group in the clusterer.
640  * @private
641  */
642 ClusterManager.prototype.cluster = function(type) {
643     var precision = this.getPrecision();
644     if (typeof type === "undefined") {
645         var clusters = this.clusters[precision];
646         for (var type in clusters) {
647             this.cluster(type);
648         }
649         return;
650     }
651     if (typeof this.markers[type] === "undefined") return; //no markers to cluster
652     if (typeof this.markers[type]["cluster"] !== "undefined") {
653         for (var i = 0, marker; marker = this.markers[type]["cluster"][i]; i++) {
654             marker.setVisible(false);
655         }
656     }
657     this.markers[type]["cluster"] = [];
658     this.cluster_meta[type]["count"]["cluster"] = 0;
659     var clusters = this.clusters;
660     if (this.opts.cluster_by_distance) this.combineClustersByDistance(type);
661     for (var boxStr in clusters[precision][type]) {
662         //visualize the boxes by adding polygons to the map for debugging.
663         if (this.opts.visualize) this.boxToPolygon(boxStr).setMap(this.map);
664         var cluster = clusters[precision][type][boxStr];
665         for (var i = 0, cluster_markers = []; marker = cluster["markers"][i]; i++) {
666             var meta = this.getMarkerMeta(marker);
667             if (typeof meta.hidden === "undefined" || !meta.hidden) {
668                 cluster_markers.push(marker);
669             }
670         }
671         if (cluster_markers.length > 1) {
672             cluster["cluster"] = this.cluster_fns[type](cluster_markers, cluster["center"][0], 
673                                                         cluster["center"][1], this);
674             this.addMarker(cluster["cluster"], {
675                 type    : type,
676                 subtype : "cluster",
677                 hidden  : false
678             });
679             this.cluster_meta[type]["count"]["cluster"] += 1;
680         } else {
681             cluster["cluster"] = false;
682         }
683     }
684 };

686 /**
687  * Gets the markers of a given type and/or subtype. Returns all markers if passed no parameters.
688  *
689  * @param {string} [type] The type of the markers to return.
690  * @param {string} [subtype] The subtype of the markers to return.
691  * @param {string|boolean} [visible] Pass "all" to get markers that aren't clusters.
692                                      Pass true to get all markers that are visible and not hidden.
693  * @returns {google.maps.Marker[]} The markers of the given type.
694  */
695 ClusterManager.prototype.getMarkers = function(type, subtype, visible) {
696     var markers = [];
697     if (this.markers === {}) return []; //no markers of any type.
698     if (typeof type === "undefined") {
699         for (var type in this.markers) {
700             for (var subtype in this.markers[type]) {
701                 markers = markers.concat(this.markers[type][subtype]);
702             }
703         }
704     } else if (typeof subtype === "undefined") {
705         for (var subtype in this.markers[type]) {
706             //access all subcategories with a string.
707             markers = markers.concat(this.markers[type][subtype]); 
708         }
709     } else {
710         try {
711             markers = this.markers[type][subtype] || [];
712         } catch (err) {
713             markers = [];
714         }
715     }
716     if (typeof visible === "undefined") return markers;
717 
718     for (var i=0, final_markers=[], length=markers.length; i<length; i++) {
719         var marker = markers[i];
720         var meta = this.getMarkerMeta(marker);
721         if (visible === "all" || meta.hidden !== visible && meta.visible == visible && 
722             typeof marker !== "function" && meta.type !== "cluster") {
723             final_markers.push(marker);
724         }
725     }
726     return final_markers;
727 };

729 /**
730  * Handles any change in the map viewport. Calls updateMarkers with a timeout so it doesn't lock up 
731  * the map.
732  * @private
733  */
734 ClusterManager.prototype._onMapMoveEnd = function() {
735     var me = this;
736     if (typeof me.moveTimeout !== "undefined") {
737         clearTimeout(me.moveTimeout);
738         delete(me.moveTimeout);
739     }
740     var precision = me.zoomToPrecision(me.map.getZoom());
741     if (me.getPrecision() !== precision) {
742         me.setPrecision(precision);
743     } else {
744         me.moveTimeout = setTimeout(function() {
745             delete(me.moveTimeout);
746             me.updateMarkers();
747         }, 100);
748     }
749 };

790 /**
791  * Since clustering takes time, this sets up a delay before reclustering.
792  * 
793  * @param {string} type The type to update.
794  * @private
795  */
796 ClusterManager.prototype._lagUpdate = function(type) {
797     var me = this;
798     if (typeof this.processingTimeout !== "undefined") {
799         clearTimeout(me.processingTimeout);
800         delete(me.processingTimeout);
801     }
802     this.processingTimeout = setTimeout(function() {
803         delete(me.processingTimeout);
804         me.clear(type);
805         me.cluster(type);
806         me.updateMarkers();
807     }, 100);
808 };

810 /**
811  * This sets a cluster type to an empty state.
812  *
813  * @param {string} [type] The type to reset. If none is given, every type in the clusterer is reset.
814  */
815 ClusterManager.prototype.reset = function(type) {
816     if(typeof type === "undefined") {
817         var clusters = this.clusters[this.getPrecision()];
818         for(var type in clusters) {
819             this.reset(type);
820         }
821         return;
822     }
823     this.clear(type);
824     //this for loop should probably be a reset cluster function
825     for(var precision in this.clusters) {
826         delete(this.clusters[precision][type]);
827         this.clusters[precision][type] = {};
828     }
829     delete(this.markers[type]);
830     this.markers[type] = {};
831 };

833 /**
834  * This removes the markers from the map. Use reset if you want to actually get rid of the 
835  * markers.
836  *  
837  * @param {string} [type] The type to clear. If it is not passed, all markers managed by the 
838  * clusterer will be cleared.
839  */
840 ClusterManager.prototype.clear = function(type) {
841     var markers = this.getMarkers(type);
842     for(var i=0, length=markers.length; i<length; i++) { 
843         var marker = markers[i];
844         marker.setMap(null);
845         this.getMarkerMeta(marker).visible = false;
846     }
847     if (typeof type !== "undefined") {

849     } else {
850         for (var item in this.cluster_meta) {
851             this.cluster_meta[item]["count"]["visible"] = 0;
852         }
853     }

866 /**
867  * Updates the markers on the map based on the current viewport with padding.
868  * @private
869  */
870 ClusterManager.prototype.updateMarkers = function() {
871     var precision = this.getPrecision();
872     var currentBounds = this.map.getBounds();
873     var cluster = this.clusters[precision];
874     for (var type in cluster) {
875         var type_cluster = cluster[type];
876         for (var box in type_cluster) {
877             var cluster_box = type_cluster[box];
878             var cluster_box_meta = this.getMarkerMeta(cluster_box["cluster"]);
879             if (this.boxInBounds(box, currentBounds, this.opts.padding)) {
880                 if (cluster_box["cluster"]) {
881                     if (!cluster_box_meta.hidden && !cluster_box_meta.visible) {
882                         for(var i=0, length=cluster_box["markers"].length; i<length; i++) { 
883                             var marker = cluster_box["markers"][i];
884                             this.getMarkerMeta(marker).visible = true;
885                         }
886                         cluster_box["cluster"].setMap(this.map);
887                         cluster_box["cluster"].setVisible(true);
888                         cluster_box_meta.visible = true;
889                         this.cluster_meta[type]["count"]["visible"] += 1;
890                     }
891                 } else {
892                     var marker = cluster_box["markers"][0];
893                     var meta = this.getMarkerMeta(marker);
894                     if (!meta.hidden && !meta.visible) {
895                         marker.setMap(this.map);
896                         marker.setVisible(true);
897                         meta.visible = true;
898                         this.cluster_meta[type]["count"]["visible"] += 1;
899                     }
900                 }
901             } else {
902                 if (cluster_box["cluster"]) {
903                     cluster_box["cluster"].setVisible(false);
904                     if (cluster_box_meta.visible) this.cluster_meta[type]["count"]["visible"] -= 1;
905                     cluster_box_meta.visible = false;
906                 } else {
907                     for(var i=0, length=cluster_box["markers"].length; i<length; i++) { 
908                         var marker = cluster_box["markers"][i];
909                         var meta = this.getMarkerMeta(marker);
910                         marker.setVisible(false);
911                         if (meta.visible) this.cluster_meta[type]["count"]["visible"] -= 1;
912                         meta.visible = false;
913                     }
914                 }
915             }
916         }
917     }
918 };
