1 /****
  2  * @name ClusterManager
  3  * @version 1.0
  4  * @author Marcus Ulrich
  5  * @fileoverview
  6  * This library creates and manages clusters for v2 Google Maps API. It does two
  7  * things to make maps with large numbers of markers more useable: 1) Puts markers in 
  8  * close proximity to each other based on zoom level into clusters, 2) Only adds 
  9  * markers in the current viewport (and optional padding) to the map.
 10  * * <b>How it works</b>:<br/>
 11  * The manager sets up a dictionary for clusters and a dictionary for markers. Every
 12  * marker that's added to the manager has a string created based on it's latitude,
 13  * longitude, and zoom level and that's used to add it to the cluster dictionary. Nearby
 14  * markers will hash to the same string so nothing has to be calculated. Nearby clusters
 15  * are then combined.
 16  * Markers can be added with optional type and subtypes so subsets of markers can be shown 
 17  * and hidden.
 18  * The function used to create the clusters is stored in its own dictionary and this function
 19  * can be overridden for greater control of the look of the clusters.
 20  *
 21 ****/

 27 /**
 28  * Creates a new Cluster Manager for clustering markers on a V2 Google map.
 29  *
 30  * @constructor
 31  * @param {GMap2} map The map that the markers should be added to.
 32  * @param {Object} [opts] Same as for resetManager.
 33  */
 34 ClusterManager = function(map, opts) {
 35     this.map = map;
 36     this.resetManager(opts);
 37     this.currentPrecision = this.zoomToPrecision(this.map.getZoom());
 38     GEvent.bind(map, "moveend", this, this._onMapMoveEnd);
 39 }

 41 /**
 42  * Sets the marker and clusters back to the inital state.
 43  *
 44  * @param {Object} [opts.cluster_defs] Defines a precision for each zoom level. 
 45  * @param {String} [opts.icon_color] Sets the default icon color. 
 46  * @param {Number} [opts.padding] The amount of padding in pixels where markers not in the viewport will
 47  *                                still be added to the map.
 48  * @param {Boolean} [opts.visualize] For debugging. Will put a box around each cluster with at least
 49  *                                   one marker.
 50  */
 51 ClusterManager.prototype.resetManager = function(opts) {
 52     this.markers = {}; //hold markers by type, then subtype.
 53     this.clusters = {}; //define clusters by precision, type, then geobox.
 54     this.cluster_fns = {}; //store cluster function for building the cluster markers.
 55     this.cluster_meta = {}; //marker counts, etc
 56     this.cluster_precisions = {"-2": {zoom: {min: 0, max: 2}},
 57                                "-1": {zoom: {min: 3, max: 4}},
 58                                   0: {zoom: {min: 5, max: 7}}, 
 59                                   1: {zoom: {min: 8, max: 9}}, 
 60                                   2: {zoom: {min: 10, max: 11}},
 61                                   3: {zoom: {min: 12, max: 12}},
 62                                   4: {zoom: {min: 13, max: 13}},
 63                                   5: {zoom: {min: 14, max: 100}}}; //define the zoom level for each precision.
 64     opts = ClusterManager.applyDefaults({cluster_defs : this.cluster_precisions, 
 65                                          padding      : 200, 
 66                                          visualize    : false, 
 67                                          icon_color   : "00CC00"}, opts);
 68     this.opts = opts;
 69     this.cluster_precisions = opts.cluster_defs;
 70 }

 72 /**
 73  * Gets a zero padded string as a hash for a latitude, longitude, or other number.
 74  *
 75  * @param {Number} num The number to be hashed
 76  * @param {Number} precision The number of places to round to after the decimal or before w/negatives.
 77  * @param {Number} mod Mod the number to reduce the number of possible boxes.
 78  * @param {Number} tens The number of places before the decimal. This includes space for negatives.
 79  * @return {String} The geohash of the input number.
 80  *
 81  **/
 82 ClusterManager.prototype.getZeroPaddedString = function(num, precision, mod, tens) {
 83     num = (Math.round(num * Math.pow(10, precision)) - Math.round(num * Math.pow(10, precision)) % mod)/Math.pow(10, precision);
 84     var numString = num + "";
 85     if(num > 0) numString = "0" + numString; //include space for negatives.
 86     var length = tens + 1; //include space for negative or zero.
 87     if(precision > 0) {
 88         length += precision + 1; //include space for decimal point.
 89     } else {
 90         length += precision; //no decimal
 91     }
 92     var tensLength = (numString.indexOf(".") === -1) ? numString.length:numString.indexOf(".");
 93 
 94     //pad the front with zeros.
 95     while(tensLength < tens+1) {
 96          numString = "0" + numString;
 97          tensLength += 1;
 98     }
 99     //pad the back with zeros.
100     while(numString.length < length) {
101         if(numString.length == tens+1 && precision > 0) numString += ".";
102         else numString += "0";
103     }
104     //chop off the fat. only matters for negative precision.
105     numString = numString.substr(0, length);
106     return numString;
107 }

109 /**
110  * This gets the geobox surrounding a specified latitude and longitude.
111  *
112  * @param {Number} lat The latitude to get the surrounding geobox of.
113  * @param {Number} lng The longitude to get the surrounding geobox of.
114  * @param {Number} precision The precision level for the geobox. Higher precisions are geographically smaller areas.
115  * @return {String} A string representing the geobox. 
116  */
117 ClusterManager.prototype.getGeoBox = function(lat, lng, precision) {
118     return this.getZeroPaddedString(lat, precision, 1, 2) + this.getZeroPaddedString(lng, precision, 1, 3);
119 }

121 /**
122  * Given a geobox, this returns the bounds on it's range. The inverse of getGeoBox.
123  * @param {String} box_str A string representing the geobox.
124  * @return {Object} GLatLngBounds representing the bounds on the geobox. 
125  */
126 ClusterManager.prototype.boxToLatLngBounds = function(box_str) {
127     var precision, lat, lng, latLength=3;
128     precision = this.boxGetPrecision(box_str);
129     if(precision <= 0) {
130         lat = parseFloat(box_str.substr(0, latLength+precision));
131     } else if(precision > 0) {
132         latLength += 1; //add space for decimal.
133     }
134 
135     lat = parseFloat(box_str.substr(0, latLength+precision));
136     var lngBox = box_str.substr(latLength+precision);
137     if(lngBox.indexOf("-") !== -1) {
138         lng = parseFloat(lngBox.substr(lngBox.indexOf("-")));
139     } else {
140         lng = parseFloat(lngBox);
141     }
142 
143     if(precision <= 0) {
144         lat = lat * Math.pow(10, Math.abs(precision));
145         lng = lng * Math.pow(10, Math.abs(precision));
146     }
147 
148     var maxLat = lat + Math.pow(10, -precision)/2;
149     var maxLng = lng + Math.pow(10, -precision)/2;
150     var minLat = lat - Math.pow(10, -precision)/2;
151     var minLng = lng - Math.pow(10, -precision)/2;
152 
153     return new GLatLngBounds(new GLatLng(minLat, minLng), new GLatLng(maxLat, maxLng));
154 }

156 /**
157  * Derives the precision from a geobox string.
158  *
159  * @param {String} box_str The geobox to find the precision of.
160  * @return {Number} The derived precision of the geobox.
161  */
162 ClusterManager.prototype.boxGetPrecision = function(box_str) {
163     var precision = (box_str.length - 9)/2;
164     if(precision < 0) precision += 1; 
165     return precision;
166 }

168 /**
169  * Gets the boxes surrounding the given box in a "plus sign" shape and only returns boxes that have at least one marker.
170  *
171  * @param {String} box_str The geobox to find the neighbors of.
172  * @param {String} type The type of the geobox to find the neighbors of.
173  * @return {Array} The strings for the geoboxes with at least one marker neighboring the input geobox.
174  */
175 ClusterManager.prototype.getNeighborBoxes = function(box_str, type) {
176     var bounds = this.boxToLatLngBounds(box_str);
177     var precision = this.boxGetPrecision(box_str);
178 
179     var boxString1 = this.getGeoBox(bounds.getSouthWest().lat() + 0.0001, bounds.getSouthWest().lng() - 0.0001, precision);
180     var boxString2 = this.getGeoBox(bounds.getSouthWest().lat() - 0.0001, bounds.getSouthWest().lng() + 0.0001, precision);
181     var boxString3 = this.getGeoBox(bounds.getNorthEast().lat() + 0.0001, bounds.getNorthEast().lng() - 0.0001, precision);
182     var boxString4 = this.getGeoBox(bounds.getNorthEast().lat() - 0.0001, bounds.getNorthEast().lng() + 0.0001, precision);
183     var boxStrings = [boxString1, boxString2, boxString3, boxString4];
184 
185     for(var i=0, neighbors=[], boxString; boxString=boxStrings[i]; i++) {
186         if(typeof this.clusters[precision][type][boxString] !== "undefined") {
187             neighbors.push(boxString);
188         }
189     }
190 
191     return neighbors;
192 }

194 /**
195  * Given a geobox, this returns a GPolygon covering the box's bounds.
196  * @param {String} box_str A string representing the geobox.
197  * @param {String} opts.strokeColor The stroke color of the polygon.
198  * @param {String} opts.strokeWeight 
199  * @param {String} opts.strokeOpacity
200  * @param {String} opts.fillColor Color of the inside of the polygon
201  * @param {String} opts.fillOpacity
202  * @return {Object} GPolygon covering the box's bounds.
203  */
204 ClusterManager.prototype.boxToPolygon = function(box_str, opts) {
205     opts = ClusterManager.applyDefaults({strokeColor   : "#f33f00", 
206                                          strokeWeight  : 5,
207                                          strokeOpacity : 1,
208                                          fillColor     : "#ff0000",
209                                          fillOpacity   : 0.2}, opts);
210     var bounds = this.boxToLatLngBounds(box_str);
211     var ne = bounds.getNorthEast();
212     var sw = bounds.getSouthWest();
213     var polygon = new GPolygon([ne,new GLatLng(ne.lat(), sw.lng()), sw, new GLatLng(sw.lat(), ne.lng()), ne],
214                                opts.strokeColor, opts.strokeWeight, opts.strokeOpacity, opts.fillColor, opts.fillOpacity);
215     return polygon;
216 }

218 /**
219  * Tests whether a geobox touches a given bounds. Padding expands the range of the bounds based on viewport pixels.
220  *
221  * @param {String} box_str A string representing the geobox.
222  * @param {Object} bounds A GLatLngBounds object covering the bounds to be tested.
223  * @param {Number} [padding] Number of pixels to expand the bounds. 
224  * @return {Boolean} Returns true if any part of the geobox touches the bounds expanded by the padding.
225  */
226 ClusterManager.prototype.boxInBounds = function(box_str, bounds, padding) {
227     var newBounds = new GLatLngBounds(bounds.getSouthWest(), bounds.getNorthEast());
228     if(typeof padding !== "undefined") {
229         var nePixel = this.map.fromLatLngToDivPixel(bounds.getNorthEast());
230         var swPixel = this.map.fromLatLngToDivPixel(bounds.getSouthWest());
231         var newNE = this.map.fromDivPixelToLatLng(new GPoint(nePixel.x + padding, nePixel.y - padding));
232         var newSW = this.map.fromDivPixelToLatLng(new GPoint(swPixel.x - padding, swPixel.y + padding));
233         newBounds.extend(newNE);
234         newBounds.extend(newSW);
235     }
236 
237     var boxBounds = this.boxToLatLngBounds(box_str);
238 
239     if(newBounds.contains(boxBounds.getNorthEast()) || newBounds.contains(boxBounds.getSouthWest())) return true;
240     else return false;
241 }

244 /**
245  * Add a single marker to the map. Stores an associate array for looking for marker types
246  * so we can cluster by type. Doesn't build clusters or add them to the map. Each
247  * marker can have an opt type and subtype to cluster by. 
248  *
249  * @param {Marker} marker The GMarker to add. 
250  * @param {String} [opts.type] A string that is used to sort which markers to cluster.
251  * @param {String} [opts.subtype] A string that is used to show/hide subsets of markers of a given type.
252  * @param {Boolean} [opts.hidden] Set true to make a marker disappear from the map even if it's in the viewport.
253  * @param {Boolean} [opts.visible] Set true if the marker is visible in the viewport. 
254  * @param {String} [opts.summary] The summary text that appears in the cluster's infowindow. Clicking on the
255  *                              text opens the markers infowindow.
256  */
257 ClusterManager.prototype.addMarker = function(marker, opts) {
258     var me=this;
259  
260     if(typeof opts === "undefined") opts = this.getMarkerMeta(marker);
261     //Set when the marker is visible in the viewport and not hidden.
262     //Set when we want to hide the marker even if it's in the viewport.
263     var defaults = {type    : "generic",
264                     subtype : "generic",
265                     hidden  : true,
266                     visible : false};
267     opts = ClusterManager.applyDefaults(defaults, opts);
268     var type=opts.type, subtype=opts.subtype;
269 
270     //if this is the first marker of the type, save the cluster function.
271     if(typeof this.markers[type] === "undefined") {
272         this.markers[type] = {};
273         this.cluster_meta[type] = {count: {total:0, visible:0, cluster:0}}; 
274     }
275     if(typeof this.cluster_fns[type] === "undefined") {
276         this.cluster_fns[type] = function(marker_list, center, precision, map) {return me.createClusterMarker(marker_list, center, precision, map);};
277     }
278     //if this is the first marker of the subtype, set up an empty array to save it in.
279     if(typeof this.markers[type][subtype] === "undefined") {
280         this.markers[type][subtype] = [];
281     }
282     this.markers[type][subtype].push(marker);
283     if(subtype !== "cluster") {
284         this.cluster_meta[type]["count"]["total"] += 1; 
285         this.addToCluster(marker, type, this.currentPrecision);
286     }
287     if(typeof opts.summary === "undefined") {
288         var capType = opts.type.charAt(0).toUpperCase() + opts.type.slice(1);
289         opts.summary = typeof marker.getTitle() === "undefined" ? capType + " marker "+this.count(opts.type, "total"):marker.getTitle();
290     }
291     this.setMarkerMeta(marker, opts);
292 }

304 /**
305  * Adds a marker to a cluster object. Does not create the cluster markers.
306  *
307  * @param {Marker} marker The marker to add. 
308  * @param {String} type The type of the marker to add. This will be used to form cluster groups. If no
309  *                      type is given it is assigned type "generic".
310  * @param {Number} precision The precision to cluster at.
311  * @param {String} [box_str] Force a marker into a particular geobox rather than its default one.
312  */
313 ClusterManager.prototype.addToCluster = function(marker, type, precision, box_str) {
314     var clusters = this.clusters;
315     var markerLL = marker.getLatLng();
316     var markerLat = markerLL.lat();
317     var markerLng = markerLL.lng();
318     if(typeof clusters[precision] === "undefined") {
319         clusters[precision] = {};
320     }
321     if(typeof clusters[precision][type] === "undefined") {
322         clusters[precision][type] = {};
323     }
324     var cluster = clusters[precision][type];
325     if(typeof box_str === "undefined") box_str = this.getGeoBox(markerLat, markerLng, precision);
326 
327     if(typeof cluster[box_str] !== "undefined") { 
328         cluster[box_str]["markers"].push(marker);
329         var length = cluster[box_str]["markers"].length;
330         var lat = ((length-1)/length)*cluster[box_str]["center"][0] + markerLat/length;
331         var lng = ((length-1)/length)*cluster[box_str]["center"][1] + markerLng/length;
332         cluster[box_str]["center"] = [lat, lng];
333     } else {
334         cluster[box_str] = {cluster: false, markers: [marker], center: [markerLat, markerLng]};
335     }
336 }

338 /**
339  * Removes a marker from a cluster and resets the cluster boxes properties.
340  *
341  * @param {Marker} marker The GMarker to remove.
342  * @param {String} box_str The geobox to remove the marker from.
343  */
344 ClusterManager.prototype.removeFromCluster = function(marker, box_str) {
345     var precision = this.boxGetPrecision(box_str);
346     var type = this.getMarkerMeta(marker).type;
347     var box = this.clusters[precision][type][box_str];
348 
349     if(box["markers"].length === 1) {
350         delete(this.clusters[precision][type][box_str]);        
351     } else if(box["markers"].length > 1) {
352         for(var i=0, new_markers=[], center_lat=0, center_lng=0, test_marker; test_marker=box["markers"][i]; i++) {
353             if(test_marker !== marker) {
354                 new_markers.push(test_marker);
355                 center_lat = center_lat + test_marker.getLatLng().lat();
356                 center_lng = center_lng + test_marker.getLatLng().lng();
357             }
358         }
359         center_lat = center_lat/new_markers.length;
360         center_lng = center_lng/new_markers.length;
361         box["center"] = [center_lat, center_lng];
362         box["markers"] = new_markers;
363         box["cluster"] = false;
364         this.clusters[precision][type][box_str] = box;
365     }
366 }

368 /**
369  * This takes two geoboxes and puts all the markers into the one with more markers or the first one.
370  * 
371  * @param {String} box_str1 First box to combine.
372  * @param {String} box_str2 Second box to combine.
373  * @param {String} type Type of the boxes since this can't be derived.
374  */
375 ClusterManager.prototype.combineBoxes = function(box_str1, box_str2, type) {
376     var precision = this.boxGetPrecision(box_str1);
377     if(this.clusters[precision][type][box_str1]["markers"].length < this.clusters[precision][type][box_str2]["markers"].length) {
378         var temp = box_str1;
379         box_str1 = box_str2;
380         box_str2 = temp;
381     }
382 
383     var length = this.clusters[precision][type][box_str2]["markers"].length;
384     for(var i=length-1, marker; i>=0; i--) {
385         marker = this.clusters[precision][type][box_str2]["markers"][i];
386         this.removeFromCluster(marker, box_str2);
387         this.addToCluster(marker, type, precision, box_str1);
388     }
389 }

391 /**
392  * This builds the actual clusters but does not set up the cluster dictionary.
393  *
394  * @param {String} [type] The group type to cluster. If none is given, this sets up the clusters for every group in the
395  * clusterer.
396  */
397 ClusterManager.prototype.cluster = function(type) {
398     var precision = this.currentPrecision;
399 
400     if(typeof type === "undefined") {
401         var clusters = this.clusters[precision];
402         for(var type in clusters) {
403             this.cluster(type);
404         }
405         return;
406     }
407     var iw = this.map.getInfoWindow();
408     var iw_hidden = iw.isHidden();
409     if(typeof this.markers[type] === "undefined") return; //no markers to cluster
410 
411     if(typeof this.markers[type]["cluster"] !== "undefined") {
412         for(var i=0, marker; marker=this.markers[type]["cluster"][i]; i++) this.map.removeOverlay(marker);
413     }
414     this.markers[type]["cluster"] = [];
415     this.cluster_meta[type]["count"]["cluster"] = 0;
416     var clusters = this.clusters;
417 
418     for(var boxStr in clusters[precision][type]) {
419         var neighbors = this.getNeighborBoxes(boxStr, type);
420         var distance = Math.pow(10, -precision);
421         for(var j=0, neighborStr; neighborStr=neighbors[j]; j++) {
422             if(Math.abs(clusters[precision][type][boxStr]["center"][0] - clusters[precision][type][neighborStr]["center"][0]) < distance &&
423                Math.abs(clusters[precision][type][boxStr]["center"][1] - clusters[precision][type][neighborStr]["center"][1]) < distance) {
424                 this.combineBoxes(boxStr, neighborStr, type);
425                 break;
426             }
427         }
428     }
429 
430     for(var boxStr in clusters[precision][type]) {
431         //visualize the boxes by adding polygons to the map for debugging.
432         if(this.opts.visualize) this.map.addOverlay(this.boxToPolygon(boxStr));
433 
434         var cluster = clusters[precision][type][boxStr];
435         for(var i=0, cluster_markers=[]; marker=cluster["markers"][i]; i++) {
436             var meta = this.getMarkerMeta(marker);
437             if(typeof meta.hidden === "undefined" || !meta.hidden) {
438                 cluster_markers.push(marker); 
439             }
440         }
441         if(cluster_markers.length > 1) { 
442             cluster["cluster"] = this.cluster_fns[type](cluster_markers, cluster["center"], precision, this.map);
443             this.addMarker(cluster["cluster"], {type: type, subtype: "cluster", hidden: false});
444             this.cluster_meta[type]["count"]["cluster"] += 1;
445         } else {
446             cluster["cluster"] = false;
447         }
448     }
449 
450     if(!iw_hidden) {
451         this.map.openInfoWindow(iw.getPoint(), iw.getContentContainers()[0]);
452     }
453 }

456 /**
457  * Gets the markers of a given type and/or subtype. Returns all markers if passed no parameters.
458  *
459  * @param {String} [type] The type of the markers to return.
460  * @param {String} [subtype] The subtype of the markers to return.
461  * @param {String | Boolean} [visible] Pass "all" to get markers that aren't clusters.
462                                        Pass true to get all markers that are visible and not hidden.
463  * @return {Array} The markers of the given type.
464  */
465 ClusterManager.prototype.getMarkers = function(type, subtype, visible) {
466     var markers = [];
467     if(this.markers === {}) return []; //no markers of any type.
468     if(typeof type === "undefined") {
469         for(var type in this.markers) {
470             for(var subtype in this.markers[type]) {
471                 markers = markers.concat(this.markers[type][subtype]);
472             }
473         }
474     } else if(typeof subtype === "undefined") {
475         for(var subtype in this.markers[type]) {
476             markers = markers.concat(this.markers[type][subtype]); //access all subcategories with a string.
477         }
478     } else { 
479         try {
480             markers = this.markers[type][subtype] || [];
481         } catch(err) {
482             markers = [];
483         }
484     }
485     if(typeof visible === "undefined") return markers;
486     for(var i=0, marker, final_markers=[]; marker=markers[i]; i++) {
487         var meta = this.getMarkerMeta(marker);
488         if(visible === "all" || meta.hidden !== visible && meta.visible == visible && typeof marker !== "function" && meta.type !== "cluster") {
489             final_markers.push(marker);
490         }
491     }
492     return final_markers;
493 }

495 /**
496  * Handles any change in the map viewport. Calls updateMarkers with a timeout so it doesn't
497  * lock up the map.
498  */
499 ClusterManager.prototype._onMapMoveEnd = function() {
500     var me=this;
501     var precision = this.zoomToPrecision(this.map.getZoom());
502     if(this.currentPrecision !== precision) {
503         this.changePrecision(precision);
504     } else {
505         setTimeout(function() {me.updateMarkers();}, 10);
506     }
507 }

531  * Does the actual showing or hiding.
532  */
533 ClusterManager.prototype._showHide = function(type, subtype, hide) {
534     var markers = this.getMarkers(type, subtype);
535     for(var i=0, marker; marker=markers[i]; i++) {
536         this.getMarkerMeta(marker).hidden = hide;
537     }

542 /**
543  * Since clustering takes time, this sets up a delay before reclustering.
544  */
545 ClusterManager.prototype._lagUpdate = function(type) {
546     var me = this;
547     if(typeof this.processingTimeout !== "undefined") { clearTimeout(me.processingTimeout); 
548                                                         delete(me.processingTimeout);};
549     this.processingTimeout = setTimeout(function() { delete(me.processingTimeout);
550                                                      me.clear(type);
551                                                      me.cluster(type);
552                                                      me.updateMarkers();
553                                                      }, 300);
554 }

558  * This sets a cluster type to an empty state.
559  *
560  * @param {String} type The group type to reset.
561  */
562 ClusterManager.prototype.reset = function(type) {
563     this.clear(type);
564     //this for loop should probably be a reset cluster function
565     for(var precision in this.clusters) {
566         delete(this.clusters[precision][type]);
567         this.clusters[precision][type] = {};
568     }
569     delete(this.markers[type]);
570     this.markers[type] = {};
571 }

574 /**
575  * This removes all the markers from the map.
576  */
577 ClusterManager.prototype.clear = function(type) {
578     var markers = this.getMarkers(type);
579 
580     for(var i=0, marker; marker=markers[i]; i++) {
581         this.map.removeOverlay(marker);
582         this.getMarkerMeta(marker).visible = false;
583     }
584     if(typeof type !== "undefined") {
585         this.cluster_meta[type]["count"]["visible"] = 0;
586     } else {
587         for(var item in this.cluster_meta) {
588             this.cluster_meta[item]["count"]["visible"] = 0;
589         }
590     }
591 
592 }

595 /**
596  * Convert a map zoom level to a precision.
597  *
598  * @param {Number} zoom_level The map zoom level
599  * @return {Number} The precision of the input zoom level. Null if the precision is not in a range.
600  */
601 ClusterManager.prototype.zoomToPrecision = function(zoom_level) {
602     for(var precision in this.cluster_precisions) {
603         var ranges = this.cluster_precisions[parseInt(precision)];
604         if(zoom_level >= ranges.zoom.min && zoom_level <= ranges.zoom.max) {
605             return parseInt(precision);
606         }
607     }
608     return null;
609 }

612 /**
613  * Called whenever the viewport change also requires a change in precision level for the clusterer.
614  * To speed up clustering and reduce memory, only the clusters for the current precision are calculated
615  * so changing the precision may take extra time to calculate clusters at the new precision.
616  * @param {Number} The precision level to set as the current precision.
617  */
618 ClusterManager.prototype.changePrecision = function(precision) {
619     this.currentPrecision = precision;
620     this.clear();
621     if(typeof this.clusters[precision] === "undefined") {
622         var markers = this.getMarkers();
623         for(var i=0, marker; marker=markers[i]; i++) {
624             if(this.getMarkerMeta(marker).subtype !== "cluster") {
625                 this.addToCluster(marker, this.getMarkerMeta(marker).type, precision);
626             }
627         }
628     }
629     this.cluster();
630     this.updateMarkers();
631 }

633 /**
634  * Updates the markers on the map based on the current viewport with padding.
635  */
636 ClusterManager.prototype.updateMarkers = function() {
637     var me=this;
638     var precision = this.currentPrecision;
639     var currentBounds = this.map.getBounds();
640     var cluster = this.clusters[precision];
641     for(var type in cluster) {
642         var type_cluster = cluster[type];
643         for(var box in type_cluster) {
644             var cluster_box = type_cluster[box];
645             var cluster_box_meta = this.getMarkerMeta(cluster_box["cluster"]);
646             if(this.boxInBounds(box, currentBounds, this.opts.padding)) {
647                 if(cluster_box["cluster"]) {
648                     if(!cluster_box_meta.hidden && !cluster_box_meta.visible) {
649                         for(var i=0, marker; marker=cluster_box["markers"][i]; i++) {
650                             this.getMarkerMeta(marker).visible = true;
651                         }
652                         me.map.addOverlay(cluster_box["cluster"]);
653                         cluster_box_meta.visible = true;
654                         this.cluster_meta[type]["count"]["visible"] += 1;
655                     }
656                 } else {
657                     var marker = cluster_box["markers"][0];
658                     var meta = this.getMarkerMeta(marker);
659                     if(!meta.hidden && !meta.visible) {
660                         me.map.addOverlay(marker);
661                         meta.visible = true;
662                         this.cluster_meta[type]["count"]["visible"] += 1;
663                     }
664                 }
665             } else {
666                 if(cluster_box["cluster"]) {
667                     me.map.removeOverlay(cluster_box["cluster"]);
668                     if(cluster_box_meta.visible) this.cluster_meta[type]["count"]["visible"] -= 1;
669                     cluster_box_meta.visible = false;
670                 } else {
671                     for(var i=0, marker; marker=cluster_box["markers"][i]; i++) {
672                         var meta = this.getMarkerMeta(marker);
673                         me.map.removeOverlay(marker);
674                         if(meta.visible) this.cluster_meta[type]["count"]["visible"] -= 1;
675                         meta.visible = false;
676                     }
677                 }
678             }
679         }
680     }
681 }

721 ClusterManager.prototype.createClusterIcon = function(number, precision) {
722     var size = ((number + "").length - 1)*6 + 24;
723 
724     if(precision > 1) {
725         var iconOpts = {"image"    : 'http://chart.apis.google.com/chart?cht=d&chdp=mapsapi&chl=pin%27i\\%27[' + 
726                                       number + '%27-2%27f\\hv%27a\\]h\\]o\\' + this.opts.icon_color + '%27fC\\000000%27tC\\000000%27eC\\Lauto%27f\\&ext=.png',

728     } else {
729         var imageMap = [];
730         var polyNumSides = 8;
731         var polySideLength = 360/polyNumSides;
732         var polyRadius = Math.min(size, size)/2;
733         for(var a=0; a<(polyNumSides+1); a++) {
734             var aRad = polySideLength * a * (Math.PI/180);
735             var pixelX = polyRadius + polyRadius * Math.cos(aRad);
736             var pixelY = polyRadius + polyRadius * Math.sin(aRad);
737             imageMap.push(parseInt(pixelX), parseInt(pixelY));
738         } 
739         var iconOpts = { "shadow"           : new GSize(0, 0),
740                          "iconSize"         : new GSize(size, size), 
741                          "iconAnchor"       : new GPoint(size/2, size/2), 
742                          "infoWindowAnchor" : new GPoint(size/2, size/2),
743                          "imageMap"         : imageMap,
744                          "image"            : "http://chart.apis.google.com/chart?cht=it&chs=" + size + "x" + size + "&chco=" + this.opts.icon_color + ",000000ff,ffffff01&chl=" + 
745                                                number + "&chx=000000,0&chf=bg,s,00000000&ext=.png"};
746     }
747     return this.createIcon(iconOpts);

751 /**
752  * A free function for creating cluster markers.
753  */
754 ClusterManager.prototype.createClusterMarker = function(marker_list, center, precision, map) {
755     var me=this;
756     var htmlEl = document.createElement("div");
757     htmlEl.style.width = "400px";
758     function markerClickClosure(marker) {
759         return function() {
760             map.closeInfoWindow(); 
761             var meta = me.getMarkerMeta(marker);
762             if(typeof meta.fn !== "undefined") {
763                 meta.fn();
764             } else {
765                 map.openInfoWindowHtml(marker.getLatLng(), meta.content, {maxTitle: meta.maxTitle, maxContent: meta.maxContent});
766             }
767         };
768     }
769     for(var i=0, marker; marker=marker_list[i]; i++) {
770         var markerSpan = document.createElement("span");
771         markerSpan.innerHTML = '<b>' + this.getMarkerMeta(marker).summary + '</b><br>'
772         markerSpan.onclick = markerClickClosure(marker);
773         markerSpan.style.color = "#334499";
774         markerSpan.style.cursor = "pointer";
775         htmlEl.appendChild(markerSpan);
776         if(i>=9) break;
777     }
778 
779     if(marker_list.length > 10) {
780         htmlEl.appendChild(document.createTextNode((marker_list.length - 10) + " more markers in this area. Zoom in for details."));
781     }
782 
783     if(precision <= 1) var zIndexProcess = function(marker, b) {return marker.getZIndex*marker_list.length;};
784     else var zIndexProcess = undefined;
785     
786     var marker = this.createMarker(center[0], center[1], 
787                                          {"title"         : marker_list.length + " markers",
788                                           "content"       : htmlEl,
789                                           "summary"       : marker_list.length + " markers",
790                                           "icon"          : this.createClusterIcon(marker_list.length, precision),
791                                           "zIndexProcess" : zIndexProcess
792                                           });
793     return marker;
794 }

797 /**
798  * A free function for creating GIcons.
799  */
800 ClusterManager.prototype.createIcon = function(opts) {
801     if(typeof opts === "undefined") opts = {};
802     if(typeof opts.width === "undefined") opts.width = 32;
803     if(typeof opts.height === "undefined") opts.height = 32;
804     var width=opts.width, height=opts.height;
805 
806     if(typeof opts.strokeColor === "undefined") opts.strokeColor = "#000000";
807     if(typeof opts.cornerColor === "undefined") opts.cornerColor = "#ffffff"; 
808     var baseUrl = "http://chart.apis.google.com/chart?cht=mm";
809     var iconUrl = baseUrl + "&chs=" + width + "x" + height + "&chco=" + opts.cornerColor.replace("#", "") + "," +
810                   this.opts.icon_color + "," + opts.strokeColor.replace("#", "") + "&ext=.png"; 
811 
812     opts = ClusterManager.applyDefaults({"image"            : iconUrl,
813                                          "iconSize"         : new GSize(width, height), 
814                                          "iconAnchor"       : new GPoint(width/2, height), 
815                                          "infoWindowAnchor" : new GPoint(width/2, Math.floor(height/12)),
816                                          "transparent"      : iconUrl + "&chf=a,s,ffffff11&ext=.png",
817                                          "printImage"       : iconUrl + "&chof=gif",
818                                          "mozPrintImage"    : iconUrl + "&chf=bg,s,ECECD8" + "&chof=gif",
819                                          "shadowSize"       : new GSize(Math.floor(width*1.6), height),
820                                          "imageMap"         : [width/2, height,
821                                                               (7/16)*width, (5/8)*height,
822                                                               (5/16)*width, (7/16)*height,
823                                                               (7/32)*width, (5/16)*height,
824                                                               (5/16)*width, (1/8)*height,
825                                                               (1/2)*width, 0,
826                                                               (11/16)*width, (1/8)*height,
827                                                               (25/32)*width, (5/16)*height,
828                                                               (11/16)*width, (7/16)*height,
829                                                               (9/16)*width, (5/8)*height]}, opts);
830     for(var i=0; i<opts.imageMap.length; i++) {
831         opts.imageMap[i] = parseInt(opts.imageMap[i]);
832     } 
833 
834     var icon = new GIcon(G_DEFAULT_ICON);
835     for(var index in opts) {
836         icon[index] = opts[index];
837     }
838     return icon;
839 }

842 /**
843  * A free function for creating GMarkers.
844  */
845 ClusterManager.prototype.createMarker = function(lat, lng, opts) {
846     var icon = this.createIcon();
847     var defaults = {
848            "icon"          : icon,
849            "content"       : "Marker",
850            "maxTitle"      : undefined,
851            "maxContent"    : undefined,
852            "fn"            : undefined,
853            "zIndexProcess" : undefined
854         };
855     opts = ClusterManager.applyDefaults(defaults, opts);
856 
857     var marker = new GMarker(new GLatLng(lat, lng), {title:opts.title, icon:opts.icon, zIndexProcess:opts.zIndexProcess});
858 
859     if(typeof opts.fn === "undefined") {
860         var fn = function() { marker.openInfoWindowHtml(opts.content, {maxTitle: opts.maxTitle, maxContent: opts.maxContent});};
861     } else {
862         var fn = opts.fn;
863     }
864 
865     this.setMarkerMeta(marker, opts);
866 
867     GEvent.addListener(marker, "click", fn);
868     return marker;
869 }

872 /**
873  * Tool for setting defaults.
874  */
875 ClusterManager.applyDefaults = function(defaults, opts) {
876     if(typeof defaults !== "object") return opts;
877     if(typeof opts !== "object") return defaults;
878 
879     for(var index in defaults) {
880         if(typeof opts[index] === "undefined") {
881             opts[index] = defaults[index];
882         }
883     }
884     return opts;
885 }
